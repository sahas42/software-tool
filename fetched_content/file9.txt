// AspireBridge Database Schema Extensions
// This file contains all new tables for the book donation platform

import Database from 'better-sqlite3';

// ============= NEW TABLE CREATION QUERIES =============

// 1. Book Listings Table - stores book donation sets
export const createBookListingsTable = `
  CREATE TABLE IF NOT EXISTS book_listings (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    donor_id INTEGER NOT NULL,
    title TEXT NOT NULL,
    description TEXT,
    exam_type TEXT NOT NULL,
    subjects TEXT NOT NULL,
    condition TEXT NOT NULL,
    year_used TEXT,
    location_city TEXT NOT NULL,
    location_state TEXT NOT NULL,
    status TEXT NOT NULL DEFAULT 'available',
    allocated_to INTEGER,
    images TEXT,
    created_at TEXT NOT NULL,
    updated_at TEXT NOT NULL,
    FOREIGN KEY (donor_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (allocated_to) REFERENCES users(id) ON DELETE SET NULL
  )
`;

// 2. Recipient Profiles Table - stores student verification data
export const createRecipientProfilesTable = `
  CREATE TABLE IF NOT EXISTS recipient_profiles (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL UNIQUE,
    full_name TEXT NOT NULL,
    phone_number TEXT NOT NULL,
    date_of_birth TEXT NOT NULL,
    current_class TEXT NOT NULL,
    target_exam TEXT NOT NULL,
    tenth_percentage REAL NOT NULL,
    tenth_board TEXT NOT NULL,
    tenth_marksheet_file TEXT,
    school_id INTEGER,
    school_name TEXT NOT NULL,
    school_address TEXT NOT NULL,
    school_city TEXT NOT NULL,
    school_state TEXT NOT NULL,
    school_pincode TEXT NOT NULL,
    verification_status TEXT NOT NULL DEFAULT 'pending',
    verified_by INTEGER,
    verified_at TEXT,
    rejection_reason TEXT,
    created_at TEXT NOT NULL,
    updated_at TEXT NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (school_id) REFERENCES schools(id) ON DELETE SET NULL,
    FOREIGN KEY (verified_by) REFERENCES users(id) ON DELETE SET NULL
  )
`;

// 3. Book Requests Table - tracks which recipients requested which books
export const createBookRequestsTable = `
  CREATE TABLE IF NOT EXISTS book_requests (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    listing_id INTEGER NOT NULL,
    recipient_id INTEGER NOT NULL,
    status TEXT NOT NULL DEFAULT 'pending',
    priority_score REAL,
    admin_notes TEXT,
    created_at TEXT NOT NULL,
    updated_at TEXT NOT NULL,
    FOREIGN KEY (listing_id) REFERENCES book_listings(id) ON DELETE CASCADE,
    FOREIGN KEY (recipient_id) REFERENCES users(id) ON DELETE CASCADE,
    UNIQUE(listing_id, recipient_id)
  )
`;

// 4. Schools Table - verified school database
export const createSchoolsTable = `
  CREATE TABLE IF NOT EXISTS schools (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    address TEXT NOT NULL,
    city TEXT NOT NULL,
    state TEXT NOT NULL,
    pincode TEXT NOT NULL,
    board TEXT,
    verified BOOLEAN DEFAULT 0,
    created_at TEXT NOT NULL,
    UNIQUE(name, city, pincode)
  )
`;

// 5. Shipments Table - tracks book shipments
export const createShipmentsTable = `
  CREATE TABLE IF NOT EXISTS shipments (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    listing_id INTEGER NOT NULL,
    donor_id INTEGER NOT NULL,
    recipient_id INTEGER NOT NULL,
    school_id INTEGER NOT NULL,
    tracking_number TEXT,
    courier_service TEXT,
    estimated_cost REAL,
    actual_cost REAL,
    funded_by_platform BOOLEAN DEFAULT 0,
    status TEXT NOT NULL DEFAULT 'pending',
    shipped_at TEXT,
    delivered_at TEXT,
    created_at TEXT NOT NULL,
    updated_at TEXT NOT NULL,
    FOREIGN KEY (listing_id) REFERENCES book_listings(id) ON DELETE CASCADE,
    FOREIGN KEY (donor_id) REFERENCES users(id) ON DELETE SET NULL,
    FOREIGN KEY (recipient_id) REFERENCES users(id) ON DELETE SET NULL,
    FOREIGN KEY (school_id) REFERENCES schools(id) ON DELETE SET NULL
  )
`;

// 6. Notifications Table - in-app notifications
export const createNotificationsTable = `
  CREATE TABLE IF NOT EXISTS notifications (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id INTEGER NOT NULL,
    type TEXT NOT NULL,
    title TEXT NOT NULL,
    message TEXT NOT NULL,
    link TEXT,
    read BOOLEAN DEFAULT 0,
    created_at TEXT NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
  )
`;

// ============= INDEXES FOR PERFORMANCE =============

export const createAspireBridgeIndexes = `
  CREATE INDEX IF NOT EXISTS idx_book_listings_donor ON book_listings(donor_id);
  CREATE INDEX IF NOT EXISTS idx_book_listings_status ON book_listings(status);
  CREATE INDEX IF NOT EXISTS idx_book_listings_exam ON book_listings(exam_type);
  CREATE INDEX IF NOT EXISTS idx_book_listings_location ON book_listings(location_city, location_state);
  
  CREATE INDEX IF NOT EXISTS idx_recipient_profiles_user ON recipient_profiles(user_id);
  CREATE INDEX IF NOT EXISTS idx_recipient_profiles_verification ON recipient_profiles(verification_status);
  CREATE INDEX IF NOT EXISTS idx_recipient_profiles_phone ON recipient_profiles(phone_number);
  
  CREATE INDEX IF NOT EXISTS idx_book_requests_listing ON book_requests(listing_id);
  CREATE INDEX IF NOT EXISTS idx_book_requests_recipient ON book_requests(recipient_id);
  CREATE INDEX IF NOT EXISTS idx_book_requests_status ON book_requests(status);
  
  CREATE INDEX IF NOT EXISTS idx_schools_city ON schools(city, state);
  CREATE INDEX IF NOT EXISTS idx_schools_verified ON schools(verified);
  
  CREATE INDEX IF NOT EXISTS idx_shipments_listing ON shipments(listing_id);
  CREATE INDEX IF NOT EXISTS idx_shipments_donor ON shipments(donor_id);
  CREATE INDEX IF NOT EXISTS idx_shipments_recipient ON shipments(recipient_id);
  CREATE INDEX IF NOT EXISTS idx_shipments_status ON shipments(status);
  
  CREATE INDEX IF NOT EXISTS idx_notifications_user ON notifications(user_id);
  CREATE INDEX IF NOT EXISTS idx_notifications_read ON notifications(read);
`;

// ============= MIGRATION FUNCTIONS =============

export function runAspireBridgeMigration(db: Database.Database): void {
    console.log('Running AspireBridge database migration...');

    // 1. Add user_type column to users table if it doesn't exist
    const userTableInfo = db.prepare("PRAGMA table_info(users)").all() as Array<{ name: string }>;
    const hasUserType = userTableInfo.some(col => col.name === 'user_type');

    if (!hasUserType) {
        console.log('Adding user_type column to users table...');
        db.exec("ALTER TABLE users ADD COLUMN user_type TEXT DEFAULT 'donor'");
    }

    // 2. Rename donations table to monetary_donations if not already renamed
    const tables = db.prepare("SELECT name FROM sqlite_master WHERE type='table'").all() as Array<{ name: string }>;
    const hasDonationsTable = tables.some(t => t.name === 'donations');
    const hasMonetaryDonationsTable = tables.some(t => t.name === 'monetary_donations');

    if (hasDonationsTable && !hasMonetaryDonationsTable) {
        console.log('Renaming donations table to monetary_donations...');
        db.exec('ALTER TABLE donations RENAME TO monetary_donations');
    }

    // 3. Create new tables
    console.log('Creating new AspireBridge tables...');
    db.exec(createBookListingsTable);
    db.exec(createRecipientProfilesTable);
    db.exec(createBookRequestsTable);
    db.exec(createSchoolsTable);
    db.exec(createShipmentsTable);
    db.exec(createNotificationsTable);

    // 4. Create indexes
    console.log('Creating indexes...');
    db.exec(createAspireBridgeIndexes);

    console.log('AspireBridge migration completed successfully!');
}
