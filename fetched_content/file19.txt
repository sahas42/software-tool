// Quick script to update donator1 account to donor type
import db from '../database';

try {
    const username = 'donator1';

    // Check if user exists
    const user = db.prepare('SELECT * FROM users WHERE username = ?').get(username) as any;

    if (!user) {
        console.log(`❌ User '${username}' not found.`);
        process.exit(1);
    }

    console.log('Current user details:');
    console.log(`  Username: ${user.username}`);
    console.log(`  Email: ${user.email}`);
    console.log(`  Role: ${user.role}`);
    console.log(`  User Type: ${user.user_type || 'NULL (not set)'}`);
    console.log('');

    // Update user_type to donor
    const stmt = db.prepare('UPDATE users SET user_type = ? WHERE username = ?');
    stmt.run('donor', username);

    console.log(`✅ Successfully updated user_type to 'donor' for ${username}`);
    console.log('');
    console.log('Next steps:');
    console.log('  1. Logout from the web app');
    console.log('  2. Login again');
    console.log('  3. You will now see the "+ Create Listing" button in the navbar!');

} catch (error) {
    console.error('❌ Error:', error);
    process.exit(1);
}
