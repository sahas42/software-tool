// DonationPage: Main donation page with form and list of donations
// Enhanced with user authentication - logged-in users get auto-filled form and can only delete their own donations

// Import React hooks for state management and side effects
import { useState, useEffect } from 'react';

// Import the useAuth hook to access current user data
import { useAuth } from '../contexts/AuthContext';

// Import the original App.css for styling
import '../App.css';

// TypeScript interface defining the structure of a Donation object
// Updated to include user_id field for ownership tracking
interface Donation {
  id: number;             // Unique identifier from database
  user_id: number | null; // ID of user who made donation (null for anonymous/legacy)
  name: string;           // Donor's name
  email: string;          // Donor's email address
  amount: number;         // Donation amount (as number, not string)
  timestamp: string;      // When donation was made (formatted date/time)
}

// Base URL for our backend API
const API_URL = 'http://localhost:3001/api';

// DonationPage functional component
function DonationPage() {
  // Get current user from AuthContext
  // user will be null if not logged in, or a User object if logged in
  const { user } = useAuth();

  // State for form input fields
  const [name, setName] = useState('');
  const [email, setEmail] = useState('');
  const [amount, setAmount] = useState('');

  // State for the list of donations
  const [donations, setDonations] = useState<Donation[]>([]);

  // loading: boolean indicating if a request is in progress
  const [loading, setLoading] = useState(false);

  // error: string containing error message to display to user
  const [error, setError] = useState('');

  // useEffect to auto-fill form when user logs in
  // Runs whenever 'user' changes (login, logout, etc.)
  useEffect(() => {
    // If user is logged in, auto-fill name and email fields
    if (user) {
      setName(user.username);  // Use username as name
      setEmail(user.email);    // Use email from user profile
    } else {
      // If user logs out, clear the form
      setName('');
      setEmail('');
    }
  }, [user]); // Dependency array - run this effect when user changes

  // useEffect to fetch donations when component mounts
  // Empty dependency array means this runs once on mount
  useEffect(() => {
    fetchDonations();
  }, []);

  // Async function to fetch all donations from the backend
  const fetchDonations = async () => {
    try {
      // Make a GET request to the backend API
      // credentials: 'include' sends session cookie (needed for auth)
      const response = await fetch(`${API_URL}/donations`, {
        credentials: 'include'
      });

      // Check if the response was successful
      if (!response.ok) throw new Error('Failed to fetch donations');

      // Parse the JSON response body
      const data = await response.json();

      // Update the donations state with the fetched data
      setDonations(data);
    } catch (err) {
      // If anything goes wrong, log and show error
      console.error('Error fetching donations:', err);
      setError('Failed to load donations');
    }
  };

  // Form submission handler
  const handleSubmit = async (e: React.FormEvent) => {
    // Prevent the default form submission behavior (page reload)
    e.preventDefault();

    // Clear any previous error messages
    setError('');

    // Client-side validation - check if all fields have values
    if (!name.trim() || !email.trim() || !amount.trim()) {
      setError('Please fill in all fields');
      return;
    }

    // Convert amount from string to number
    const amountNum = parseFloat(amount);

    // Validate the amount is a valid positive number
    if (isNaN(amountNum) || amountNum <= 0) {
      setError('Please enter a valid amount');
      return;
    }

    // Set loading to true - disables form during submission
    setLoading(true);

    try {
      // Make a POST request to add the new donation
      const response = await fetch(`${API_URL}/donations`, {
        method: 'POST',                        // HTTP method for creating resources
        headers: {
          'Content-Type': 'application/json',  // Sending JSON data
        },
        credentials: 'include',                // Send session cookie (associates donation with user if logged in)
        body: JSON.stringify({                 // Convert object to JSON string
          name: name.trim(),
          email: email.trim(),
          amount: amountNum,
        }),
      });

      // Check if the request was successful
      if (!response.ok) {
        const errorData = await response.json();
        throw new Error(errorData.error || 'Failed to add donation');
      }

      // Parse the response to get the newly created donation object
      const newDonation = await response.json();

      // Add the new donation to the beginning of our donations array
      setDonations([newDonation, ...donations]);

      // Only clear amount field - keep name and email for convenience
      // (If user is logged in, these will be their username and email)
      setAmount('');
    } catch (err: any) {
      // If anything goes wrong, catch the error
      console.error('Error adding donation:', err);
      setError(err.message || 'Failed to add donation');
    } finally {
      // Set loading back to false to re-enable the form
      setLoading(false);
    }
  };

  // Handler for deleting a donation
  // Now checks if user is logged in and owns the donation
  const handleDelete = async (id: number) => {
    // Show a confirmation dialog to prevent accidental deletions
    if (!confirm('Are you sure you want to delete this donation?')) {
      return; // User cancelled
    }

    try {
      // Make a DELETE request to the backend
      // Backend will check if user owns this donation
      const response = await fetch(`${API_URL}/donations/${id}`, {
        method: 'DELETE',          // HTTP method for deleting resources
        credentials: 'include',    // Send session cookie for authorization check
      });

      // Check if deletion was successful
      if (!response.ok) {
        // If failed, show error message
        const errorData = await response.json();
        throw new Error(errorData.error || 'Failed to delete donation');
      }

      // Remove the deleted donation from our local state
      setDonations(donations.filter(d => d.id !== id));
    } catch (err: any) {
      // If deletion fails, log and show error
      console.error('Error deleting donation:', err);
      setError(err.message || 'Failed to delete donation');
    }
  };

  // Calculate the total of all donations
  // reduce() iterates through array, accumulating a sum
  const totalDonations = donations.reduce((sum, donation) => sum + donation.amount, 0);

  // Helper function to determine if user can delete a donation
  // Returns true if user is logged in AND (owns the donation OR is admin)
  const canDelete = (donation: Donation): boolean => {
    // If no user logged in, can't delete anything
    if (!user) return false;

    // If user is admin, can delete any donation (future feature)
    if (user.role === 'admin') return true;

    // Otherwise, can only delete if donation belongs to this user
    return donation.user_id === user.id;
  };

  // Return the JSX that defines what this component renders
  return (
    <div className="App">
      <div className="container">
        {/* Main heading */}
        <h1>Donation Form</h1>

        {/* Show logged-in user info if user is logged in */}
        {user && (
          <div style={{
            backgroundColor: '#e7f3ff',
            padding: '10px',
            borderRadius: '4px',
            marginBottom: '20px',
            textAlign: 'center'
          }}>
            Logged in as <strong>{user.username}</strong>
          </div>
        )}

        {/* Conditionally render error message if error exists */}
        {error && <div className="error-message">{error}</div>}

        {/* Donation form */}
        <form onSubmit={handleSubmit} className="donation-form">
          {/* Name input field group */}
          <div className="form-group">
            <label htmlFor="name">Name:</label>
            <input
              type="text"
              id="name"
              value={name}                                  // Controlled component - value from state
              onChange={(e) => setName(e.target.value)}     // Update state on change
              placeholder="Enter your name"
              disabled={loading}                            // Disable during submission
            />
          </div>

          {/* Email input field group */}
          <div className="form-group">
            <label htmlFor="email">Email:</label>
            <input
              type="email"
              id="email"
              value={email}                                 // Controlled component
              onChange={(e) => setEmail(e.target.value)}
              placeholder="Enter your email"
              disabled={loading}
            />
          </div>

          {/* Amount input field group */}
          <div className="form-group">
            <label htmlFor="amount">Amount ($):</label>
            <input
              type="number"
              id="amount"
              value={amount}                                // Controlled component
              onChange={(e) => setAmount(e.target.value)}
              placeholder="Enter amount"
              step="0.01"                                   // Allow decimals
              min="0"                                       // Don't allow negative
              disabled={loading}
            />
          </div>

          {/* Submit button */}
          <button type="submit" className="submit-button" disabled={loading}>
            {/* Conditional text based on loading state */}
            {loading ? 'Submitting...' : 'Submit Donation'}
          </button>
        </form>

        {/* Conditionally render donations section only if there are donations */}
        {donations.length > 0 && (
          <div className="donations-section">
            <h2>Donations Received</h2>

            {/* Total donations display */}
            <div className="total">
              Total Donations: ${totalDonations.toFixed(2)}
            </div>

            {/* Donations table */}
            <div className="donations-table-wrapper">
              <table className="donations-table">
                {/* Table header */}
                <thead>
                  <tr>
                    <th>ID</th>
                    <th>Name</th>
                    <th>Email</th>
                    <th>Amount</th>
                    <th>Date & Time</th>
                    <th>Actions</th>
                  </tr>
                </thead>

                {/* Table body */}
                <tbody>
                  {/* Map over donations array to create a row for each donation */}
                  {donations.map((donation) => (
                    // Each row needs a unique key prop for React's reconciliation
                    <tr key={donation.id}>
                      <td>{donation.id}</td>
                      <td>{donation.name}</td>
                      <td>{donation.email}</td>
                      <td>${donation.amount.toFixed(2)}</td>
                      <td>{donation.timestamp}</td>
                      <td>
                        {/* Only show delete button if user can delete this donation */}
                        {canDelete(donation) ? (
                          <button
                            className="delete-button"
                            onClick={() => handleDelete(donation.id)}
                          >
                            Delete
                          </button>
                        ) : (
                          // Show dash if user can't delete
                          <span style={{ color: '#999' }}>-</span>
                        )}
                      </td>
                    </tr>
                  ))}
                </tbody>
              </table>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}

// Export DonationPage as the default export
export default DonationPage;
