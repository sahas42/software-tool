// AuthContext: Global authentication state management using React Context API
// This provides user authentication state and functions to all components in the app

import React, { createContext, useContext, useState, useEffect } from 'react';
import type { ReactNode } from 'react';

// Define the shape of a User object (matches backend UserPublic interface)
export interface User {
  id: number;
  username: string;
  email: string;
  role: 'admin' | 'client';
  user_type?: 'donor' | 'recipient' | 'both';  // AspireBridge user type
  profile_picture: string | null;
  created_at: string;
}

// Define the shape of the AuthContext
// This includes the user state, loading state, and authentication functions
interface AuthContextType {
  user: User | null;              // Current logged-in user (null if not logged in)
  loading: boolean;               // Loading state (true while checking auth status)
  login: (username: string, password: string) => Promise<void>;    // Login function
  register: (username: string, email: string, password: string, userType?: 'donor' | 'recipient' | 'both') => Promise<void>; // Register function
  logout: () => Promise<void>;    // Logout function
  updateUser: (user: User) => void; // Update user state (for profile changes)
}

// Create the context with undefined as initial value
// We use undefined to force usage of the useAuth hook (which provides type safety)
const AuthContext = createContext<AuthContextType | undefined>(undefined);

// Custom hook to access the AuthContext
// This provides type-safe access to authentication state and functions
export const useAuth = () => {
  // Get the context value
  const context = useContext(AuthContext);

  // If context is undefined, it means useAuth was called outside of AuthProvider
  // This is a programming error, so we throw an error
  if (!context) {
    throw new Error('useAuth must be used within an AuthProvider');
  }

  return context;
};

// AuthProvider component that wraps the app and provides auth state
// All children components can access auth state via useAuth hook
export const AuthProvider: React.FC<{ children: ReactNode }> = ({ children }) => {
  // State to store the current user (null if not logged in)
  const [user, setUser] = useState<User | null>(null);

  // State to track if we're currently checking authentication status
  // This is true on initial load while we check if user has a valid session
  const [loading, setLoading] = useState(true);

  // Base URL for API requests
  const API_URL = 'http://localhost:3001';

  // Check if user is already logged in when component mounts
  // This is called once when the app first loads
  useEffect(() => {
    checkAuth();
  }, []); // Empty dependency array means this runs once on mount

  // Function to check if user has a valid session
  // Calls the /api/auth/me endpoint which returns user if session is valid
  const checkAuth = async () => {
    try {
      // Make GET request to check current authentication status
      // credentials: 'include' is CRITICAL - it sends the session cookie
      const response = await fetch(`${API_URL}/api/auth/me`, {
        credentials: 'include'
      });

      // If response is OK (200), user is logged in
      if (response.ok) {
        const data = await response.json();
        setUser(data.user);
      } else {
        // If response is not OK (401), user is not logged in
        setUser(null);
      }
    } catch (error) {
      // If request fails (network error, server down), assume not logged in
      console.error('Error checking auth:', error);
      setUser(null);
    } finally {
      // Whether success or failure, we're done loading
      setLoading(false);
    }
  };

  // Login function: Authenticate user with username and password
  const login = async (username: string, password: string) => {
    // Make POST request to login endpoint
    const response = await fetch(`${API_URL}/api/auth/login`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      // credentials: 'include' sends and receives cookies (session cookie)
      credentials: 'include',
      // Send username and password as JSON
      body: JSON.stringify({ username, password }),
    });

    // If login failed (401 = wrong credentials, 400 = validation error)
    if (!response.ok) {
      const error = await response.json();
      // Throw error with message from backend
      throw new Error(error.error || 'Login failed');
    }

    // Login successful - get user data from response
    const data = await response.json();
    // Update user state with logged-in user
    setUser(data.user);
  };

  // Register function: Create new user account
  const register = async (username: string, email: string, password: string, userType: 'donor' | 'recipient' | 'both' = 'donor') => {
    // Make POST request to register endpoint
    const response = await fetch(`${API_URL}/api/auth/register`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      credentials: 'include',
      // Send registration data as JSON including user_type
      body: JSON.stringify({ username, email, password, user_type: userType }),
    });

    // If registration failed (400 = validation error or duplicate username/email)
    if (!response.ok) {
      const error = await response.json();
      // Backend returns either 'error' (single message) or 'errors' (array of messages)
      throw new Error(error.error || error.errors?.join(', ') || 'Registration failed');
    }

    // Registration successful - user is automatically logged in
    const data = await response.json();
    setUser(data.user);
  };

  // Logout function: End user session
  const logout = async () => {
    try {
      // Make POST request to logout endpoint
      // This destroys the session on the server
      await fetch(`${API_URL}/api/auth/logout`, {
        method: 'POST',
        credentials: 'include', // Send session cookie to be destroyed
      });
    } catch (error) {
      console.error('Error during logout:', error);
    } finally {
      // Whether logout request succeeded or failed, clear user state
      // This logs out the user on the frontend
      setUser(null);
    }
  };

  // Update user state (used after profile updates)
  // This allows other components to update the user without re-fetching
  const updateUser = (updatedUser: User) => {
    setUser(updatedUser);
  };

  // The value object that will be provided to all children components
  const value = {
    user,
    loading,
    login,
    register,
    logout,
    updateUser,
  };

  // Provide the auth context to all children components
  return <AuthContext.Provider value={value}>{children}</AuthContext.Provider>;
};
