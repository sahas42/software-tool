// AdminRequests component: Manage pending book requests
// Allows admin to approve or reject book allocation requests

import React, { useState, useEffect } from 'react';
import { useAuth } from '../contexts/AuthContext';

interface BookRequest {
    id: number;
    listing_id: number;
    recipient_id: number;
    recipient_name: string;
    book_title: string;
    priority_score: number;
    status: string;
    created_at: string;
}

const AdminRequests: React.FC = () => {
    const { user } = useAuth();
    const [requests, setRequests] = useState<BookRequest[]>([]);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState('');
    const [processingId, setProcessingId] = useState<number | null>(null);

    const API_URL = 'http://localhost:3001';

    useEffect(() => {
        if (user?.role === 'admin') {
            fetchPendingRequests();
        }
    }, [user]);

    const fetchPendingRequests = async () => {
        try {
            setLoading(true);
            const response = await fetch(`${API_URL}/api/admin/requests`, {
                credentials: 'include'
            });

            if (!response.ok) {
                throw new Error('Failed to fetch requests');
            }

            const data = await response.json();
            setRequests(data);
            setError('');
        } catch (err) {
            setError(err instanceof Error ? err.message : 'Failed to load requests');
        } finally {
            setLoading(false);
        }
    };

    const handleApprove = async (requestId: number) => {
        const notes = prompt('Enter any notes for this approval (optional):');

        try {
            setProcessingId(requestId);
            const response = await fetch(`${API_URL}/api/admin/requests/${requestId}/approve`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                credentials: 'include',
                body: JSON.stringify({ notes: notes || '' }),
            });

            if (!response.ok) {
                const errorData = await response.json();
                throw new Error(errorData.error || 'Failed to approve request');
            }

            // Refresh the list
            fetchPendingRequests();
        } catch (err) {
            alert(err instanceof Error ? err.message : 'Failed to approve request');
        } finally {
            setProcessingId(null);
        }
    };

    const handleReject = async (requestId: number) => {
        const reason = prompt('Enter rejection reason:');
        if (!reason) return;

        try {
            setProcessingId(requestId);
            const response = await fetch(`${API_URL}/api/admin/requests/${requestId}/reject`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                credentials: 'include',
                body: JSON.stringify({ reason }),
            });

            if (!response.ok) {
                const errorData = await response.json();
                throw new Error(errorData.error || 'Failed to reject request');
            }

            // Refresh the list
            fetchPendingRequests();
        } catch (err) {
            alert(err instanceof Error ? err.message : 'Failed to reject request');
        } finally {
            setProcessingId(null);
        }
    };

    if (user?.role !== 'admin') {
        return (
            <div style={{ maxWidth: '1200px', margin: '50px auto', padding: '20px' }}>
                <h1>Access Denied</h1>
                <p>You must be an admin to access this page.</p>
            </div>
        );
    }

    return (
        <div style={{ maxWidth: '1200px', margin: '50px auto', padding: '20px' }}>
            <h1>Pending Book Requests</h1>
            <p style={{ color: '#666', marginBottom: '30px' }}>
                Review and approve book allocation requests from verified recipients
            </p>

            {loading && <p>Loading pending requests...</p>}

            {error && (
                <div style={{
                    padding: '15px',
                    backgroundColor: '#ffebee',
                    color: '#c62828',
                    borderRadius: '4px',
                    marginBottom: '20px'
                }}>
                    {error}
                </div>
            )}

            {!loading && !error && requests.length === 0 && (
                <div style={{
                    textAlign: 'center',
                    padding: '60px 20px',
                    backgroundColor: '#f5f5f5',
                    borderRadius: '8px'
                }}>
                    <h3>No pending requests</h3>
                    <p style={{ color: '#666' }}>All book requests have been processed</p>
                </div>
            )}

            {!loading && !error && requests.length > 0 && (
                <div style={{ display: 'flex', flexDirection: 'column', gap: '20px' }}>
                    {requests.map((request) => (
                        <div
                            key={request.id}
                            style={{
                                border: '1px solid #ddd',
                                borderRadius: '8px',
                                padding: '20px',
                                backgroundColor: 'white'
                            }}
                        >
                            <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'start', marginBottom: '15px' }}>
                                <div>
                                    <h3 style={{ margin: '0 0 5px 0' }}>{request.book_title}</h3>
                                    <p style={{ margin: 0, color: '#666', fontSize: '14px' }}>
                                        Request ID: {request.id} | Listing ID: {request.listing_id}
                                    </p>
                                </div>
                                <div style={{ textAlign: 'right' }}>
                                    <span style={{
                                        display: 'block',
                                        padding: '6px 12px',
                                        backgroundColor: '#ffc107',
                                        color: '#000',
                                        borderRadius: '4px',
                                        fontSize: '14px',
                                        marginBottom: '5px'
                                    }}>
                                        Pending
                                    </span>
                                    <span style={{
                                        display: 'block',
                                        padding: '6px 12px',
                                        backgroundColor: '#007bff',
                                        color: 'white',
                                        borderRadius: '4px',
                                        fontSize: '14px'
                                    }}>
                                        Score: {request.priority_score.toFixed(2)}
                                    </span>
                                </div>
                            </div>

                            <div style={{
                                display: 'grid',
                                gridTemplateColumns: 'repeat(auto-fit, minmax(250px, 1fr))',
                                gap: '15px',
                                marginBottom: '20px'
                            }}>
                                <div>
                                    <strong>Recipient:</strong> {request.recipient_name}
                                </div>
                                <div>
                                    <strong>Recipient ID:</strong> {request.recipient_id}
                                </div>
                                <div>
                                    <strong>Requested:</strong> {new Date(request.created_at).toLocaleDateString()}
                                </div>
                            </div>

                            <div style={{ display: 'flex', gap: '10px' }}>
                                <button
                                    onClick={() => handleApprove(request.id)}
                                    disabled={processingId === request.id}
                                    style={{
                                        flex: 1,
                                        padding: '12px',
                                        backgroundColor: processingId === request.id ? '#ccc' : '#28a745',
                                        color: 'white',
                                        border: 'none',
                                        borderRadius: '4px',
                                        cursor: processingId === request.id ? 'not-allowed' : 'pointer',
                                        fontWeight: 'bold'
                                    }}
                                >
                                    ✓ Approve & Allocate
                                </button>
                                <button
                                    onClick={() => handleReject(request.id)}
                                    disabled={processingId === request.id}
                                    style={{
                                        flex: 1,
                                        padding: '12px',
                                        backgroundColor: processingId === request.id ? '#ccc' : '#dc3545',
                                        color: 'white',
                                        border: 'none',
                                        borderRadius: '4px',
                                        cursor: processingId === request.id ? 'not-allowed' : 'pointer',
                                        fontWeight: 'bold'
                                    }}
                                >
                                    ✗ Reject
                                </button>
                            </div>
                        </div>
                    ))}
                </div>
            )}
        </div>
    );
};

export default AdminRequests;
