// AdminVerifications component: Manage pending recipient profile verifications
// Allows admin to approve or reject student profiles

import React, { useState, useEffect } from 'react';
import { useAuth } from '../contexts/AuthContext';

interface RecipientProfile {
    id: number;
    user_id: number;
    full_name: string;
    phone_number: string;
    current_class: string;
    target_exam: string;
    tenth_percentage: number;
    tenth_board: string;
    school_name: string;
    school_city: string;
    school_state: string;
    verification_status: string;
    created_at: string;
}

const AdminVerifications: React.FC = () => {
    const { user } = useAuth();
    const [profiles, setProfiles] = useState<RecipientProfile[]>([]);
    const [loading, setLoading] = useState(true);
    const [error, setError] = useState('');
    const [processingId, setProcessingId] = useState<number | null>(null);

    const API_URL = 'http://localhost:3001';

    useEffect(() => {
        if (user?.role === 'admin') {
            fetchPendingProfiles();
        }
    }, [user]);

    const fetchPendingProfiles = async () => {
        try {
            setLoading(true);
            const response = await fetch(`${API_URL}/api/admin/profiles/pending`, {
                credentials: 'include'
            });

            if (!response.ok) {
                throw new Error('Failed to fetch profiles');
            }

            const data = await response.json();
            setProfiles(data);
            setError('');
        } catch (err) {
            setError(err instanceof Error ? err.message : 'Failed to load profiles');
        } finally {
            setLoading(false);
        }
    };

    const handleVerify = async (profileId: number, approved: boolean) => {
        const reason = approved ? '' : prompt('Enter rejection reason:');
        if (!approved && !reason) return;

        try {
            setProcessingId(profileId);
            const response = await fetch(`${API_URL}/api/admin/profiles/${profileId}/verify`, {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                credentials: 'include',
                body: JSON.stringify({ approved, reason }),
            });

            if (!response.ok) {
                throw new Error('Failed to process verification');
            }

            // Refresh the list
            fetchPendingProfiles();
        } catch (err) {
            alert(err instanceof Error ? err.message : 'Failed to process verification');
        } finally {
            setProcessingId(null);
        }
    };

    if (user?.role !== 'admin') {
        return (
            <div style={{ maxWidth: '1200px', margin: '50px auto', padding: '20px' }}>
                <h1>Access Denied</h1>
                <p>You must be an admin to access this page.</p>
            </div>
        );
    }

    return (
        <div style={{ maxWidth: '1200px', margin: '50px auto', padding: '20px' }}>
            <h1>Pending Profile Verifications</h1>
            <p style={{ color: '#666', marginBottom: '30px' }}>
                Review and verify recipient profiles before they can request books
            </p>

            {loading && <p>Loading pending verifications...</p>}

            {error && (
                <div style={{
                    padding: '15px',
                    backgroundColor: '#ffebee',
                    color: '#c62828',
                    borderRadius: '4px',
                    marginBottom: '20px'
                }}>
                    {error}
                </div>
            )}

            {!loading && !error && profiles.length === 0 && (
                <div style={{
                    textAlign: 'center',
                    padding: '60px 20px',
                    backgroundColor: '#f5f5f5',
                    borderRadius: '8px'
                }}>
                    <h3>No pending verifications</h3>
                    <p style={{ color: '#666' }}>All recipient profiles have been reviewed</p>
                </div>
            )}

            {!loading && !error && profiles.length > 0 && (
                <div style={{ display: 'flex', flexDirection: 'column', gap: '20px' }}>
                    {profiles.map((profile) => (
                        <div
                            key={profile.id}
                            style={{
                                border: '1px solid #ddd',
                                borderRadius: '8px',
                                padding: '20px',
                                backgroundColor: 'white'
                            }}
                        >
                            <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'start', marginBottom: '20px' }}>
                                <div>
                                    <h3 style={{ margin: '0 0 5px 0' }}>{profile.full_name}</h3>
                                    <p style={{ margin: 0, color: '#666', fontSize: '14px' }}>
                                        User ID: {profile.user_id} | Profile ID: {profile.id}
                                    </p>
                                </div>
                                <span style={{
                                    padding: '6px 12px',
                                    backgroundColor: '#ffc107',
                                    color: '#000',
                                    borderRadius: '4px',
                                    fontSize: '14px'
                                }}>
                                    Pending
                                </span>
                            </div>

                            <div style={{
                                display: 'grid',
                                gridTemplateColumns: 'repeat(auto-fit, minmax(250px, 1fr))',
                                gap: '15px',
                                marginBottom: '20px'
                            }}>
                                <div>
                                    <strong>Phone:</strong> {profile.phone_number}
                                </div>
                                <div>
                                    <strong>Current Class:</strong> {profile.current_class}
                                </div>
                                <div>
                                    <strong>Target Exam:</strong> {profile.target_exam}
                                </div>
                                <div>
                                    <strong>10th Percentage:</strong> {profile.tenth_percentage}%
                                </div>
                                <div>
                                    <strong>10th Board:</strong> {profile.tenth_board}
                                </div>
                                <div>
                                    <strong>School:</strong> {profile.school_name}
                                </div>
                                <div>
                                    <strong>Location:</strong> {profile.school_city}, {profile.school_state}
                                </div>
                                <div>
                                    <strong>Submitted:</strong> {new Date(profile.created_at).toLocaleDateString()}
                                </div>
                            </div>

                            <div style={{ display: 'flex', gap: '10px' }}>
                                <button
                                    onClick={() => handleVerify(profile.id, true)}
                                    disabled={processingId === profile.id}
                                    style={{
                                        flex: 1,
                                        padding: '12px',
                                        backgroundColor: processingId === profile.id ? '#ccc' : '#28a745',
                                        color: 'white',
                                        border: 'none',
                                        borderRadius: '4px',
                                        cursor: processingId === profile.id ? 'not-allowed' : 'pointer',
                                        fontWeight: 'bold'
                                    }}
                                >
                                    ✓ Approve
                                </button>
                                <button
                                    onClick={() => handleVerify(profile.id, false)}
                                    disabled={processingId === profile.id}
                                    style={{
                                        flex: 1,
                                        padding: '12px',
                                        backgroundColor: processingId === profile.id ? '#ccc' : '#dc3545',
                                        color: 'white',
                                        border: 'none',
                                        borderRadius: '4px',
                                        cursor: processingId === profile.id ? 'not-allowed' : 'pointer',
                                        fontWeight: 'bold'
                                    }}
                                >
                                    ✗ Reject
                                </button>
                            </div>
                        </div>
                    ))}
                </div>
            )}
        </div>
    );
};

export default AdminVerifications;
