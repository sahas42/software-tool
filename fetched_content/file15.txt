// Database operations for Schools
// This file contains all operations for the schools table

import db from '../database';
import type { School } from '../database';

// ============= SCHOOL OPERATIONS =============

/**
 * Create a new school
 * @param schoolData - School details
 * @returns The newly created school
 */
export const createSchool = (schoolData: {
    name: string;
    address: string;
    city: string;
    state: string;
    pincode: string;
    board?: string;
}): School => {
    const now = new Date().toLocaleString();

    const stmt = db.prepare(`
    INSERT INTO schools (
      name, address, city, state, pincode, board, verified, created_at
    ) VALUES (?, ?, ?, ?, ?, ?, 0, ?)
  `);

    const info = stmt.run(
        schoolData.name,
        schoolData.address,
        schoolData.city,
        schoolData.state,
        schoolData.pincode,
        schoolData.board || null,
        now
    );

    return {
        id: info.lastInsertRowid as number,
        ...schoolData,
        board: schoolData.board || null,
        verified: false,
        created_at: now
    };
};

/**
 * Get school by ID
 * @param id - School ID
 * @returns School or undefined
 */
export const getSchoolById = (id: number): School | undefined => {
    const stmt = db.prepare('SELECT * FROM schools WHERE id = ?');
    return stmt.get(id) as School | undefined;
};

/**
 * Search schools by name or city
 * @param query - Search query
 * @returns Array of matching schools
 */
export const searchSchools = (query: string): School[] => {
    const searchPattern = `%${query}%`;
    const stmt = db.prepare(`
    SELECT * FROM schools 
    WHERE name LIKE ? OR city LIKE ?
    ORDER BY verified DESC, name ASC
    LIMIT 50
  `);
    return stmt.all(searchPattern, searchPattern) as School[];
};

/**
 * Get schools by city and state
 * @param city - City name
 * @param state - State name
 * @returns Array of schools
 */
export const getSchoolsByLocation = (city: string, state: string): School[] => {
    const stmt = db.prepare(`
    SELECT * FROM schools 
    WHERE city = ? AND state = ?
    ORDER BY verified DESC, name ASC
  `);
    return stmt.all(city, state) as School[];
};

/**
 * Verify a school (admin)
 * @param id - School ID
 * @returns true if successful
 */
export const verifySchool = (id: number): boolean => {
    const stmt = db.prepare('UPDATE schools SET verified = 1 WHERE id = ?');
    const info = stmt.run(id);
    return info.changes > 0;
};

/**
 * Get all verified schools
 * @returns Array of verified schools
 */
export const getVerifiedSchools = (): School[] => {
    const stmt = db.prepare('SELECT * FROM schools WHERE verified = 1 ORDER BY name ASC');
    return stmt.all() as School[];
};

/**
 * Check if school exists (by name, city, pincode)
 * @param name - School name
 * @param city - City
 * @param pincode - Pincode
 * @returns School if exists, undefined otherwise
 */
export const findSchool = (name: string, city: string, pincode: string): School | undefined => {
    const stmt = db.prepare(`
    SELECT * FROM schools 
    WHERE name = ? AND city = ? AND pincode = ?
  `);
    return stmt.get(name, city, pincode) as School | undefined;
};
