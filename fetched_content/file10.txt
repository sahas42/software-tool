// ============= MULTER CONFIGURATION FOR FILE UPLOADS =============

// Configure multer diskStorage to control where and how files are saved
const storage = multer.diskStorage({
  // destination: Determines where uploaded files will be stored
  destination: (req, file, cb) => {
    // Define the upload directory path (backend/uploads/profiles/)
    const uploadDir = path.join(__dirname, '../uploads/profiles');

    // Check if the directory exists, if not create it recursively
    // This prevents errors if the uploads/profiles directory doesn't exist yet
    if (!fs.existsSync(uploadDir)) {
      fs.mkdirSync(uploadDir, { recursive: true });
    }

    // Call the callback with no error (null) and the upload directory path
    cb(null, uploadDir);
  },

  // filename: Determines how the uploaded file will be named
  filename: (req, file, cb) => {
    // Get the userId from the session (set during login)
    // This ensures each user's profile picture has a unique name
    const userId = req.session.userId;

    // Extract the file extension from the original filename (e.g., ".jpg", ".png")
    const ext = path.extname(file.originalname);

    // Call the callback to set filename as "user_{userId}.{ext}"
    // Example: "user_5.jpg" for user with ID 5
    // This overwrites any previous profile picture for this user
    cb(null, `user_${userId}${ext}`);
  }
});

// Create the multer upload instance with storage configuration and validation rules
const upload = multer({
  // Use the storage configuration defined above
  storage,

  // limits: Set file size restrictions
  limits: {
    fileSize: 2 * 1024 * 1024 // 2MB maximum file size (2 * 1024 KB * 1024 bytes)
  },

  // fileFilter: Validate file types before accepting upload
  fileFilter: (req, file, cb) => {
    // Define array of allowed MIME types (image formats)
    const allowedTypes = ['image/jpeg', 'image/png', 'image/jpg'];

    // Check if the uploaded file's MIME type is in the allowed list
    if (allowedTypes.includes(file.mimetype)) {
      // File type is valid, call callback with true to accept the file
      cb(null, true);
    } else {
      // File type is invalid, call callback with error to reject the file
      cb(new Error('Only JPEG and PNG images are allowed'));
    }
  }
});

// Configure express-session for user authentication
// Sessions maintain user login state across multiple HTTP requests
app.use(session({
  // store: Where session data is stored (SQLite database in our case)
  // SqliteStore(session) wraps better-sqlite3-session-store with express-session
  store: new (SqliteStore(session))({
    // client: Pass the SQLite database connection
    client: db,

    // expired: Configuration for cleaning up old sessions
    expired: {
      // clear: Automatically delete expired sessions
      clear: true,

      // intervalMs: How often to check for and delete expired sessions
      // 900000 ms = 15 minutes
      intervalMs: 900000
    }
  }),

  // secret: Secret key used to sign the session ID cookie
  // This prevents users from tampering with session data
  // In production, this should be a long random string stored in environment variables
  secret: process.env.SESSION_SECRET || 'dev-secret-change-in-production',

  // resave: Don't save session to store if it wasn't modified during the request
  // false improves performance by avoiding unnecessary database writes
  resave: false,

  // saveUninitialized: Don't save empty sessions (before user logs in)
  // false reduces database writes and improves privacy
  saveUninitialized: false,

  // cookie: Configuration for the session cookie
  cookie: {
    // secure: Only send cookie over HTTPS (false for development, true for production)
    secure: false,

    // httpOnly: Prevent JavaScript from accessing the cookie (XSS protection)
    // This is critical for security - the cookie can only be sent in HTTP requests
    httpOnly: true,

    // sameSite: CSRF (Cross-Site Request Forgery) protection
    // 'lax' allows cookies on same-site requests and top-level navigation
    sameSite: 'lax',

    // maxAge: How long the cookie/session lasts (in milliseconds)
    // 24 * 60 * 60 * 1000 = 24 hours
    // After this time, user will be logged out
    maxAge: 24 * 60 * 60 * 1000
  }
}));

// POST /api/auth/register - Create a new user account
// This endpoint handles user registration with validation
app.post('/api/auth/register', async (req: Request, res: Response) => {
  try {
    // Extract username, email, password, and user_type from request body
    const { username, email, password, user_type } = req.body;

    // Validate the registration data using our validation function
    // This checks username format, email format, and password strength
    const errors = validateRegistration(username, email, password);

    // If there are any validation errors, return them to the user
    if (errors.length > 0) {
      return res.status(400).json({ errors });
    }

    // Check if username already exists in the database
    // This prevents duplicate usernames
    const existingUser = getUserByUsername(username);
    if (existingUser) {
      return res.status(400).json({ error: 'Username already taken' });
    }

    // Check if email already exists in the database
    // This prevents duplicate email addresses
    const existingEmail = getUserByEmail(email);
    if (existingEmail) {
      return res.status(400).json({ error: 'Email already registered' });
    }

    // Create the new user in the database with user_type
    // The password is automatically hashed by the createUser function
    const user = await createUser(username, email, password, user_type || 'donor');

    // Set session data to log the user in automatically after registration
    req.session.userId = user.id;
    req.session.username = user.username;
    req.session.role = user.role;

    // Return the public user data (without password hash)
    res.status(201).json({ user });
  } catch (error) {
    // Log any unexpected errors for debugging
    console.error('Error during registration:', error);
    res.status(500).json({ error: 'Registration failed' });
  }
});

// POST /api/auth/login - Authenticate a user
// This endpoint handles user login with username and password
app.post('/api/auth/login', async (req: Request, res: Response) => {
  try {
    // Extract username and password from request body
    const { username, password } = req.body;

    // Validate that both fields are provided
    const errors = validateLogin(username, password);
    if (errors.length > 0) {
      return res.status(400).json({ errors });
    }

    // Look up the user by username in the database
    const user = getUserByUsername(username);

    // If user doesn't exist, return error
    // Note: We use a generic message to avoid revealing whether username exists
    if (!user) {
      return res.status(401).json({ error: 'Invalid username or password' });
    }

    // Verify the password using bcrypt comparison
    // This compares the plain password with the hashed password in database
    const isValid = await verifyPassword(password, user.password_hash);

    // If password is incorrect, return error
    if (!isValid) {
      return res.status(401).json({ error: 'Invalid username or password' });
    }

    // Password is correct - create a session for the user
    req.session.userId = user.id;
    req.session.username = user.username;
    req.session.role = user.role;

    // Return public user data (including user_type)
    res.json({
      user: {
        id: user.id,
        username: user.username,
        email: user.email,
        role: user.role,
        user_type: user.user_type,  // Include user_type
        profile_picture: user.profile_picture,
        created_at: user.created_at
      }
    });
  } catch (error) {
    console.error('Error during login:', error);
    res.status(500).json({ error: 'Login failed' });
  }
});

// POST /api/auth/logout - Log out the current user
// This endpoint destroys the user's session
app.post('/api/auth/logout', (req: Request, res: Response) => {
  // Destroy the session (removes it from the database)
  req.session.destroy((err) => {
    if (err) {
      console.error('Error during logout:', err);
      return res.status(500).json({ error: 'Logout failed' });
    }

    // Clear the session cookie from the client's browser
    res.clearCookie('connect.sid'); // 'connect.sid' is the default session cookie name

    // Return success message
    res.json({ message: 'Logged out successfully' });
  });
});

// PUT /api/user/profile - Update user's profile information (username and/or email)
// This endpoint requires authentication via the requireAuth middleware
app.put('/api/user/profile', requireAuth, async (req: Request, res: Response) => {
  try {
    // Extract the fields to update from request body
    const { username, email } = req.body;

    // Create an updates object with only the fields that were provided
    const updates: { username?: string; email?: string } = {};
    if (username !== undefined) updates.username = username;
    if (email !== undefined) updates.email = email;

    // Validate the update data
    const errors = validateProfileUpdate(updates);
    if (errors.length > 0) {
      return res.status(400).json({ errors });
    }

    // If username is being changed, check if it's already taken by another user
    if (updates.username) {
      const existing = getUserByUsername(updates.username);
      // Check if username exists and belongs to a different user
      if (existing && existing.id !== req.session.userId) {
        return res.status(400).json({ error: 'Username already taken' });
      }
    }

    // If email is being changed, check if it's already taken by another user
    if (updates.email) {
      const existing = getUserByEmail(updates.email);
      // Check if email exists and belongs to a different user
      if (existing && existing.id !== req.session.userId) {
        return res.status(400).json({ error: 'Email already registered' });
      }
    }

    // Update the user profile in the database
    const updatedUser = updateUserProfile(req.session.userId!, updates);

    // If update failed (user not found)
    if (!updatedUser) {
      return res.status(404).json({ error: 'User not found' });
    }

    // If username was updated, update the session as well
    if (updates.username) {
      req.session.username = updates.username;
    }

    // Return the updated user data
    res.json({ user: updatedUser });
  } catch (error) {
    console.error('Error updating profile:', error);
    res.status(500).json({ error: 'Failed to update profile' });
  }
});

// PUT /api/user/password - Change user's password
// This endpoint requires authentication and validates the current password
app.put('/api/user/password', requireAuth, async (req: Request, res: Response) => {
  try {
    // Extract current and new passwords from request body
    const { currentPassword, newPassword } = req.body;

    // Validate the password change request
    const errors = validatePasswordChange(currentPassword, newPassword);
    if (errors.length > 0) {
      return res.status(400).json({ errors });
    }

    // Get the user from database to verify current password
    const user = getUserById(req.session.userId!);
    if (!user) {
      return res.status(404).json({ error: 'User not found' });
    }

    // Verify that the current password is correct
    const isValid = await verifyPassword(currentPassword, user.password_hash);
    if (!isValid) {
      return res.status(401).json({ error: 'Current password is incorrect' });
    }

    // Update the password in the database (it will be hashed automatically)
    const success = await updateUserPassword(req.session.userId!, newPassword);

    // If update failed
    if (!success) {
      return res.status(500).json({ error: 'Failed to update password' });
    }

    // Return success message
    res.json({ message: 'Password updated successfully' });
  } catch (error) {
    console.error('Error updating password:', error);
    res.status(500).json({ error: 'Failed to update password' });
  }
});

      return res.status(400).json({ error: error.message });

    res.status(500).json({ error: 'Failed to upload profile picture' });
  }

// DELETE /api/user/account - Delete the user's account
// This is a destructive operation and requires authentication
app.delete('/api/user/account', requireAuth, (req: Request, res: Response) => {
  try {
    // Delete the user from the database
    // Due to ON DELETE SET NULL foreign key, their donations will remain with user_id = NULL
    const success = deleteUser(req.session.userId!);

    // If deletion failed (user not found)
    if (!success) {
      return res.status(404).json({ error: 'User not found' });
    }

    // Delete the user's profile picture file if it exists
    const user = getUserById(req.session.userId!);
    if (user && user.profile_picture) {
      const picturePath = path.join(__dirname, '../uploads/profiles', user.profile_picture);
      // Use fs.unlink to delete the file (ignore errors if file doesn't exist)
      fs.unlink(picturePath, (err) => {
        if (err) console.error('Error deleting profile picture:', err);
      });
    }

    // Destroy the session (log the user out)
    req.session.destroy((err) => {
      if (err) {
        console.error('Error destroying session:', err);
      }
    });

    // Return success message
    res.json({ message: 'Account deleted successfully' });
  } catch (error) {
    console.error('Error deleting account:', error);
    res.status(500).json({ error: 'Failed to delete account' });
  }
});

// POST /api/donations - Add a new donation to the database
// This is called when user submits the donation form
// If user is logged in, their userId will be associated with the donation
app.post('/api/donations', (req: Request, res: Response) => {
  try {
    // Extract name, email, and amount from the request body
    // The frontend sends this data as JSON in the POST request
    const { name, email, amount } = req.body;

    // Validation: Check if all required fields are present
    // If any field is missing or empty, we reject the request
    if (!name || !email || !amount) {
      // Return 400 (Bad Request) status with error message
      // The 'return' prevents further code execution
      return res.status(400).json({ error: 'All fields are required' });
    }

    // Convert amount from string to number
    // parseFloat handles decimal numbers like "50.25"
    const amountNum = parseFloat(amount);

    // Validate that amount is a valid positive number
    // isNaN checks if conversion failed (e.g., user entered "abc")
    // amountNum <= 0 rejects zero or negative amounts
    if (isNaN(amountNum) || amountNum <= 0) {
      return res.status(400).json({ error: 'Invalid amount' });
    }

    // Get userId from session if user is logged in
    // req.session.userId will be undefined if user is not logged in
    const userId = req.session.userId;

    // If validation passes, add the donation to the database
    // trim() removes whitespace from beginning and end of strings
    // This prevents " John " from being stored differently than "John"
    // userId is passed as optional parameter - will be null if user not logged in
    const donation = addDonation(name.trim(), email.trim(), amountNum, userId);

    // Send back the newly created donation with status 201 (Created)
    // The frontend uses this to immediately update the UI without refetching all donations
    res.status(201).json(donation);
  } catch (error) {
    // Catch any unexpected errors (database failures, etc.)
    console.error('Error adding donation:', error);
    res.status(500).json({ error: 'Failed to add donation' });
  }
});

// DELETE /api/donations/:id - Delete a specific donation by ID
// The :id is a route parameter - if request is to /api/donations/5, then id = 5
// Users can only delete their own donations (unless admin - future feature)
app.delete('/api/donations/:id', (req: Request, res: Response) => {
  try {
    // Extract the id from the route parameters
    // req.params.id is always a string, so we convert to number with parseInt
    const id = parseInt(req.params.id);

    // Validate that the ID is a valid number
    // isNaN would be true if someone sent /api/donations/abc
    if (isNaN(id)) {
      return res.status(400).json({ error: 'Invalid ID' });
    }

    // Get userId from session if user is logged in
    // If user is logged in, we'll check ownership before deleting
    const userId = req.session.userId;

    // Attempt to delete the donation
    // If userId is provided, deleteDonation will check ownership
    // If userId is undefined (not logged in), deleteDonation won't delete anything
    // This prevents anonymous users from deleting donations
    const success = deleteDonation(id, userId);

    if (success) {
      // Deletion successful - send success message
      res.json({ message: 'Donation deleted successfully' });
    } else {
      // Deletion failed - could be because:
      // 1. Donation doesn't exist
      // 2. User doesn't own the donation
      // 3. User is not logged in
      // We use a generic message to avoid revealing which case it is
      res.status(404).json({ error: 'Donation not found or you do not have permission to delete it' });
    }
  } catch (error) {
    // Catch unexpected errors
    console.error('Error deleting donation:', error);
    res.status(500).json({ error: 'Failed to delete donation' });
  }
});

// Configure multer for marksheet uploads
const marksheetStorage = multer.diskStorage({
  destination: (req, file, cb) => {
    const uploadDir = path.join(__dirname, '../uploads/marksheets');
    if (!fs.existsSync(uploadDir)) {
      fs.mkdirSync(uploadDir, { recursive: true });
    }
    cb(null, uploadDir);
  },
  filename: (req, file, cb) => {
    const userId = req.session.userId;
    const ext = path.extname(file.originalname);
    cb(null, `marksheet_${userId}${ext}`);
  }
});

const marksheetUpload = multer({
  storage: marksheetStorage,
  limits: {
    fileSize: 10 * 1024 * 1024 // 10MB max
  },
  fileFilter: (req, file, cb) => {
    const allowedTypes = ['application/pdf', 'image/jpeg', 'image/png'];
    if (allowedTypes.includes(file.mimetype)) {
      cb(null, true);
    } else {
      cb(new Error('Only PDF, JPEG, and PNG files are allowed'));
    }
  }
});

// POST /api/requests - Create book request (verified recipient only)
app.post('/api/requests', requireAuth, (req: Request, res: Response) => {
  try {
    const { listing_id } = req.body;

    if (!listing_id) {
      return res.status(400).json({ error: 'Listing ID is required' });
    }

    // Check if recipient has verified profile
    const hasVerified = Recipients.hasVerifiedProfile(req.session.userId!);
    if (!hasVerified) {
      return res.status(403).json({ error: 'You must have a verified profile to request books' });
    }

    // Check if already requested
    const alreadyRequested = BookRequests.hasRequested(listing_id, req.session.userId!);
    if (alreadyRequested) {
      return res.status(400).json({ error: 'You have already requested this book' });
    }

    const request = BookRequests.createBookRequest(listing_id, req.session.userId!);

    // Calculate and update priority score
    const { calculateRequestPriorityScore } = require('./utils/allocation');
    const priorityScore = calculateRequestPriorityScore(req.session.userId!, listing_id);
    if (priorityScore !== null) {
      BookRequests.updatePriorityScore(request.id, priorityScore);
    }

    res.status(201).json(request);
  } catch (error) {
    console.error('Error creating book request:', error);
    res.status(500).json({ error: 'Failed to create request' });
  }
});

          approved ? 'Profile Verified' : 'Profile Rejected',

          '/recipient/profile'
        );
      }

      res.json({ message: approved ? 'Profile approved' : 'Profile rejected' });
    } else {
      res.status(404).json({ error: 'Profile not found' });
    }
  } catch (error) {
    console.error('Error verifying profile:', error);
    res.status(500).json({ error: 'Failed to verify profile' });
  }
});

app.post('/api/admin/requests/:id/approve', requireAuth, requireRole('admin'), (req: Request, res: Response) => {
  try {
    const id = parseInt(req.params.id);
    if (isNaN(id)) {
      return res.status(400).json({ error: 'Invalid ID' });
    }

    const request = BookRequests.getRequestById(id);
    if (!request) {
      return res.status(404).json({ error: 'Request not found' });
    }

    // Approve the request
    const success = BookRequests.approveRequest(id, req.body.notes);

    if (success) {
      // Allocate the listing
      BookListings.allocateListing(request.listing_id, request.recipient_id);

      // Reject other requests for this listing
      BookRequests.rejectOtherRequests(request.listing_id, id);

      // Create notification
      const listing = BookListings.getBookListingById(request.listing_id);
      if (listing) {
        Notifications.createNotification(
          request.recipient_id,
          'request_approved',
          'Book Request Approved',
          `Your request for "${listing.title}" has been approved!`,
          `/requests`
        );
      }

      res.json({ message: 'Request approved and book allocated' });
    } else {
      res.status(500).json({ error: 'Failed to approve request' });
    }
  } catch (error) {
    console.error('Error approving request:', error);
    res.status(500).json({ error: 'Failed to approve request' });
  }
});
