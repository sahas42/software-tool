// Script to update user_type for existing accounts
// Run this to fix accounts created before user_type feature was added

import db from '../database';
import readline from 'readline';

const rl = readline.createInterface({
    input: process.stdin,
    output: process.stdout
});

function question(query: string): Promise<string> {
    return new Promise(resolve => rl.question(query, resolve));
}

async function updateUserType() {
    try {
        console.log('╔════════════════════════════════════════════╗');
        console.log('║   Update User Type for Existing Account   ║');
        console.log('╚════════════════════════════════════════════╝\n');

        // Get username
        const username = await question('Enter username: ');

        // Check if user exists
        const user = db.prepare('SELECT * FROM users WHERE username = ?').get(username) as any;

        if (!user) {
            console.log(`❌ User '${username}' not found.`);
            rl.close();
            return;
        }

        console.log('\nCurrent user details:');
        console.log(`  Username: ${user.username}`);
        console.log(`  Email: ${user.email}`);
        console.log(`  Role: ${user.role}`);
        console.log(`  User Type: ${user.user_type || 'NULL (not set)'}`);

        if (user.user_type) {
            console.log(`\n⚠️  User already has user_type set to: ${user.user_type}`);
            const confirm = await question('Do you want to change it? (yes/no): ');
            if (confirm.toLowerCase() !== 'yes') {
                console.log('Cancelled.');
                rl.close();
                return;
            }
        }

        console.log('\nSelect new user type:');
        console.log('  1. donor (can donate books)');
        console.log('  2. recipient (can receive books)');
        console.log('  3. both (can donate and receive)');

        const choice = await question('Enter choice (1-3): ');

        let userType: 'donor' | 'recipient' | 'both';
        switch (choice) {
            case '1':
                userType = 'donor';
                break;
            case '2':
                userType = 'recipient';
                break;
            case '3':
                userType = 'both';
                break;
            default:
                console.log('❌ Invalid choice.');
                rl.close();
                return;
        }

        // Update user_type
        const stmt = db.prepare('UPDATE users SET user_type = ? WHERE username = ?');
        stmt.run(userType, username);

        console.log(`\n✅ Successfully updated user_type to '${userType}' for ${username}`);
        console.log('\nPlease logout and login again for changes to take effect.');

    } catch (error) {
        console.error('❌ Error:', error);
    } finally {
        rl.close();
    }
}

updateUserType();
