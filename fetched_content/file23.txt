// Allocation Algorithm Utilities
// Merit-based allocation with geographical proximity consideration

import { getRecipientProfile } from '../database/recipients';
import { getBookListingById } from '../database/bookListings';
import type { BookRequest } from '../database';

// ============= PRIORITY CALCULATION =============

/**
 * Calculate merit score from 10th percentage
 * Higher percentage = higher score (0-100 scale)
 * @param tenthPercentage - 10th grade percentage
 * @returns Merit score (0-100)
 */
export const calculateMeritScore = (tenthPercentage: number): number => {
    // Direct mapping: percentage is the score
    return Math.min(100, Math.max(0, tenthPercentage));
};

/**
 * Calculate proximity score based on location match
 * Same city = 30 points, Same state = 15 points, Different state = 0 points
 * @param donorCity - Donor's city
 * @param donorState - Donor's state
 * @param recipientCity - Recipient's city
 * @param recipientState - Recipient's state
 * @returns Proximity score (0-30)
 */
export const calculateProximityScore = (
    donorCity: string,
    donorState: string,
    recipientCity: string,
    recipientState: string
): number => {
    // Normalize for comparison (case-insensitive, trim whitespace)
    const normDonorCity = donorCity.toLowerCase().trim();
    const normDonorState = donorState.toLowerCase().trim();
    const normRecipientCity = recipientCity.toLowerCase().trim();
    const normRecipientState = recipientState.toLowerCase().trim();

    if (normDonorCity === normRecipientCity && normDonorState === normRecipientState) {
        // Same city and state
        return 30;
    } else if (normDonorState === normRecipientState) {
        // Same state, different city
        return 15;
    } else {
        // Different state
        return 0;
    }
};

/**
 * Calculate overall priority score
 * Formula: (Merit Score * 0.7) + (Proximity Score * 1.0)
 * This gives 70% weight to merit and 30% weight to proximity
 * @param meritScore - Merit score (0-100)
 * @param proximityScore - Proximity score (0-30)
 * @returns Priority score
 */
export const calculatePriorityScore = (meritScore: number, proximityScore: number): number => {
    return (meritScore * 0.7) + proximityScore;
};

/**
 * Calculate priority score for a book request
 * @param recipientId - Recipient user ID
 * @param listingId - Book listing ID
 * @returns Priority score or null if calculation fails
 */
export const calculateRequestPriorityScore = (
    recipientId: number,
    listingId: number
): number | null => {
    // Get recipient profile
    const recipientProfile = getRecipientProfile(recipientId);
    if (!recipientProfile || recipientProfile.verification_status !== 'approved') {
        return null;
    }

    // Get book listing
    const listing = getBookListingById(listingId);
    if (!listing) {
        return null;
    }

    // Calculate merit score
    const meritScore = calculateMeritScore(recipientProfile.tenth_percentage);

    // Calculate proximity score
    const proximityScore = calculateProximityScore(
        listing.location_city,
        listing.location_state,
        recipientProfile.school_city,
        recipientProfile.school_state
    );

    // Calculate and return priority score
    return calculatePriorityScore(meritScore, proximityScore);
};

// ============= REQUEST RANKING =============

/**
 * Rank book requests by priority score
 * @param requests - Array of book requests with recipient info
 * @returns Sorted array (highest priority first)
 */
export const rankRequests = (
    requests: Array<BookRequest & { priority_score: number | null }>
): Array<BookRequest & { priority_score: number | null }> => {
    return requests.sort((a, b) => {
        // Handle null scores (put them at the end)
        if (a.priority_score === null && b.priority_score === null) return 0;
        if (a.priority_score === null) return 1;
        if (b.priority_score === null) return -1;

        // Sort by priority score (descending)
        if (b.priority_score !== a.priority_score) {
            return b.priority_score - a.priority_score;
        }

        // If scores are equal, sort by request creation time (earlier first)
        return new Date(a.created_at).getTime() - new Date(b.created_at).getTime();
    });
};

/**
 * Get the top-ranked request for a listing
 * @param requests - Array of book requests
 * @returns The highest priority request or null
 */
export const getTopRequest = (
    requests: Array<BookRequest & { priority_score: number | null }>
): (BookRequest & { priority_score: number | null }) | null => {
    const ranked = rankRequests(requests);
    return ranked.length > 0 ? ranked[0] : null;
};
