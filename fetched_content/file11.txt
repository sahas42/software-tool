// Database operations for Book Listings
// This file contains all CRUD operations for the book_listings table

import db from '../database';
import type { BookListing } from '../database';

// ============= BOOK LISTING CRUD OPERATIONS =============

/**
 * Create a new book listing
 * @param donorId - ID of the user creating the listing
 * @param listingData - Book listing details
 * @returns The newly created book listing
 */
export const createBookListing = (
    donorId: number,
    listingData: {
        title: string;
        description?: string;
        exam_type: string;
        subjects: string[];  // Will be stored as JSON
        condition: string;
        year_used?: string;
        location_city: string;
        location_state: string;
        images?: string[];  // Will be stored as JSON
    }
): BookListing => {
    const now = new Date().toLocaleString();

    const stmt = db.prepare(`
    INSERT INTO book_listings (
      donor_id, title, description, exam_type, subjects, condition,
      year_used, location_city, location_state, status, images,
      created_at, updated_at
    ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, 'available', ?, ?, ?)
  `);

    const info = stmt.run(
        donorId,
        listingData.title,
        listingData.description || null,
        listingData.exam_type,
        JSON.stringify(listingData.subjects),
        listingData.condition,
        listingData.year_used || null,
        listingData.location_city,
        listingData.location_state,
        listingData.images ? JSON.stringify(listingData.images) : null,
        now,
        now
    );

    return {
        id: info.lastInsertRowid as number,
        donor_id: donorId,
        title: listingData.title,
        description: listingData.description || null,
        exam_type: listingData.exam_type,
        subjects: JSON.stringify(listingData.subjects),
        condition: listingData.condition,
        year_used: listingData.year_used || null,
        location_city: listingData.location_city,
        location_state: listingData.location_state,
        status: 'available',
        allocated_to: null,
        images: listingData.images ? JSON.stringify(listingData.images) : null,
        created_at: now,
        updated_at: now
    };
};

/**
 * Get a single book listing by ID
 * @param id - Listing ID
 * @returns The book listing or undefined if not found
 */
export const getBookListingById = (id: number): BookListing | undefined => {
    const stmt = db.prepare('SELECT * FROM book_listings WHERE id = ?');
    return stmt.get(id) as BookListing | undefined;
};

/**
 * Get all book listings with optional filters
 * @param filters - Optional filters for exam type, location, status
 * @returns Array of book listings
 */
export const getAllBookListings = (filters?: {
    exam_type?: string;
    location_city?: string;
    location_state?: string;
    status?: string;
}): BookListing[] => {
    let query = 'SELECT * FROM book_listings WHERE 1=1';
    const params: any[] = [];

    if (filters?.exam_type) {
        query += ' AND exam_type = ?';
        params.push(filters.exam_type);
    }

    if (filters?.location_city) {
        query += ' AND location_city = ?';
        params.push(filters.location_city);
    }

    if (filters?.location_state) {
        query += ' AND location_state = ?';
        params.push(filters.location_state);
    }

    if (filters?.status) {
        query += ' AND status = ?';
        params.push(filters.status);
    } else {
        // By default, only show available listings
        query += ' AND status = ?';
        params.push('available');
    }

    query += ' ORDER BY created_at DESC';

    const stmt = db.prepare(query);
    return stmt.all(...params) as BookListing[];
};

/**
 * Get all listings by a specific donor
 * @param donorId - Donor user ID
 * @returns Array of book listings
 */
export const getListingsByDonor = (donorId: number): BookListing[] => {
    const stmt = db.prepare('SELECT * FROM book_listings WHERE donor_id = ? ORDER BY created_at DESC');
    return stmt.all(donorId) as BookListing[];
};

/**
 * Update a book listing (ownership check included)
 * @param id - Listing ID
 * @param donorId - Donor user ID (for ownership verification)
 * @param updates - Fields to update
 * @returns Updated listing or null if not found/unauthorized
 */
export const updateBookListing = (
    id: number,
    donorId: number,
    updates: {
        title?: string;
        description?: string;
        condition?: string;
        status?: string;
    }
): BookListing | null => {
    const now = new Date().toLocaleString();
    const fields: string[] = [];
    const values: any[] = [];

    if (updates.title !== undefined) {
        fields.push('title = ?');
        values.push(updates.title);
    }

    if (updates.description !== undefined) {
        fields.push('description = ?');
        values.push(updates.description);
    }

    if (updates.condition !== undefined) {
        fields.push('condition = ?');
        values.push(updates.condition);
    }

    if (updates.status !== undefined) {
        fields.push('status = ?');
        values.push(updates.status);
    }

    if (fields.length === 0) {
        return getBookListingById(id) || null;
    }

    fields.push('updated_at = ?');
    values.push(now);
    values.push(id);
    values.push(donorId);

    const stmt = db.prepare(`
    UPDATE book_listings 
    SET ${fields.join(', ')} 
    WHERE id = ? AND donor_id = ?
  `);

    const info = stmt.run(...values);

    if (info.changes === 0) {
        return null;
    }

    return getBookListingById(id) || null;
};

/**
 * Delete a book listing (ownership check included)
 * @param id - Listing ID
 * @param donorId - Donor user ID (for ownership verification)
 * @returns true if deleted, false otherwise
 * @throws Error if book is allocated
 */
export const deleteBookListing = (id: number, donorId: number): boolean => {
    // First check if the book exists and belongs to the donor
    const checkStmt = db.prepare('SELECT status FROM book_listings WHERE id = ? AND donor_id = ?');
    const book = checkStmt.get(id, donorId) as { status: string } | undefined;

    if (!book) {
        return false; // Book not found or not owned by donor
    }

    // Check if book is allocated
    if (book.status === 'allocated') {
        throw new Error('Cannot delete allocated books');
    }

    // Proceed with deletion
    const stmt = db.prepare('DELETE FROM book_listings WHERE id = ? AND donor_id = ?');
    const info = stmt.run(id, donorId);
    return info.changes > 0;
};

/**
 * Allocate a listing to a recipient
 * @param listingId - Listing ID
 * @param recipientId - Recipient user ID
 * @returns true if successful, false otherwise
 */
export const allocateListing = (listingId: number, recipientId: number): boolean => {
    const now = new Date().toLocaleString();
    const stmt = db.prepare(`
    UPDATE book_listings 
    SET status = 'allocated', allocated_to = ?, updated_at = ?
    WHERE id = ? AND status = 'available'
  `);

    const info = stmt.run(recipientId, now, listingId);
    return info.changes > 0;
};

/**
 * Update listing images
 * @param listingId - Listing ID
 * @param donorId - Donor user ID (for ownership verification)
 * @param images - Array of image filenames
 * @returns true if successful, false otherwise
 */
export const updateListingImages = (
    listingId: number,
    donorId: number,
    images: string[]
): boolean => {
    const now = new Date().toLocaleString();
    const stmt = db.prepare(`
    UPDATE book_listings 
    SET images = ?, updated_at = ?
    WHERE id = ? AND donor_id = ?
  `);

    const info = stmt.run(JSON.stringify(images), now, listingId, donorId);
    return info.changes > 0;
};
